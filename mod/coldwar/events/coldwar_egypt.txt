add_namespace = coldwar_egypt

#egyptian revolution of 1952
news_event = {
	id = coldwar_egypt.0
	title = coldwar_egypt.0.t
	desc = coldwar_egypt.0.d
	picture = GFX_news_event_12
	fire_only_once = yes
	trigger = {
		tag = EGY
		is_puppet_of = ENG
		date > 1951.1.1
		has_government = neutrality
	}
	mean_time_to_happen = {
		days = 365
		modifier = {
			factor = 0.5
			country_exists = ISR
		}
		modifier = {
			factor = 0.5
			communism > 0.2
		}
		modifier = {
			factor = 2
			democracy > 0.3
		}
		modifier = {
			factor = 1.5
			has_stability > 0.5
		}
		modifier = {
			factor = 1.5
			has_stability > 0.8
		}
	}
	immediate = {
		hidden_effect = {
			ENG = {
				country_event = {
					id = coldwar_egypt.1
					days = 7
				}
			}
		}
	}
	option = {
		name = coldwar_egypt.0.a
		set_cosmetic_tag = EGY_republic
		ENG = {
			end_puppet = EGY
		}
		leave_faction = yes
		create_country_leader = {
			name = "Gamal Abdel Nasser"
			picture = "gfx/leaders/SAU/Portrait_Arabia_Generic_land_1.dds"
			ideology = despotism
			traits = {
				communism_lean
			}
		}
		add_stability = -0.1
	}
}
